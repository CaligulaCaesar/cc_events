country_event = { # Spiritualist Guru
	id = dpe_init.1001

	trigger = {
		num_owned_planets > 3
		NOT = { has_ethic = ethic_gestalt_consciousness }

			limit = { NOT = { any_owned_planet = { has_modifier = cc_religious_centre } } }

				NOT = { has_planet_flag = cc_colony_story }

	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_country_flag = dpe_recent_militarists_want_war }
		is_at_war = no
		is_subject = no
		years_of_peace = {
			value > 10
			delay = 20
		}
		OR = {
			is_militarist = yes
			AND = {
				has_faction = imperialist
				any_pop_faction = {
					is_pop_faction_type = imperialist
					support > 0.22
				}
			}
		}
		has_federation = no
		
		NOT = {
			any_country = {
				OR = {
					is_country_type = extradimensional 
					is_country_type = extradimensional_2
					is_country_type = extradimensional_3
					is_country_type = ai_empire
					is_country_type = swarm
				}
			}
		}
		
		any_relation = {
			root = {
				OR = {
					is_rival = prev
					is_hostile_to = prev
					is_domineering_to = prev
				}
				NOT = { has_truce = prev }
			}
			any_owned_planet = { is_neighbor_of = root }
			NOT = {
				relative_power = {
					who = root
					category = fleet
					value > equivalent
				}
			}
			NOT = {
				count_potential_war_participants = {
					attacker = root
					defender = this
					side = this
					limit = {
						relative_power = {
							who = prev
							category = fleet
							value > equivalent
						}
					}
					count > 0
				}
			}
		}
	}
	
	immediate = {
		random = {
			chance = 10
			country_event = { id = dpe_diplo.1001 days = 0 random = 359 }
		}
	}
}
