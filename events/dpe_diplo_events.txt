			}
			any_owned_planet = {
				is_neighbor_of = root
			}
			NOT = {

					value > equivalent
				}

				count_potential_war_participants = {
					attacker = root

					side = this

						relative_power = {
							who = prev
							category = fleet
							value > equivalent
						}
					}

		}
	}
	
	immediate = {

					factor = 1.5
					planet_stability < 50
				}
				modifier = {
					factor = 1.5
					planet_stability < 35
				}
				modifier = {
					factor = 0
					OR = {
						has_modifier = dpe_indemnity_paid

							has_modifier = dpe_indemnity_receive
						}
					}
				}
				modifier = {
					factor = 2

						limit = {
							has_citizenship_rights = yes

					}
				}

		}
	}
