	}

		years_passed > 15

		planet_stability > 65
		NOR = {
			has_planet_flag = dpe_unity_planet

			limit = { pop_has_happiness = yes }

			is_country_type = default

			}
		}
	}

	mean_time_to_happen = { years = 60 }

	immediate = {

	}

country_event = {
	id = dpe_smallpos_events.3
	hide_window = yes
	
	trigger = {
		is_country_type = default
		OR = {
			has_authority = auth_democratic
			has_authority = auth_oligarchic
		}
		NOT = { has_country_flag = dpe_recent_ruler_debate }
		exists = ruler
		any_pop_faction = {
			is_a_vanilla_faction = yes
			exists = leader
			leader = { not = { is_same_value = root.ruler } }
			faction_approval < 0.4
		}
	}
	
	mean_time_to_happen = { years = 20 }
	
	immediate = {
		if = {
			limit = {
				OR = {

				}
			}
		
			if = {
				limit = {
					any_pop_faction = {
						is_a_vanilla_faction = yes
						exists = leader
						leader = { not = { is_same_value = root.ruler } }
						faction_approval < 0.4
						support > 0.15
					}
				}
				random_pop_faction = {
					limit = {
						is_a_vanilla_faction = yes
						exists = leader
						leader = { not = { is_same_value = root.ruler } }
						faction_approval < 0.4
						support > 0.15
					}
					save_event_target_as = dpe_dissident_faction
					leader = { save_event_target_as = dpe_dissident_faction_leader }
				}
			}
			else_if = {
				limit = {
					any_pop_faction = {
						is_a_vanilla_faction = yes
						exists = leader
						leader = { not = { is_same_value = root.ruler } }
						faction_approval < 0.4
						support > 0.1

						exists = leader
						leader = { not = { is_same_value = root.ruler } }
						faction_approval < 0.4
						support > 0.1
					}
					save_event_target_as = dpe_dissident_faction
					leader = { save_event_target_as = dpe_dissident_faction_leader }
				}
			}
			else_if = {
				limit = {
					any_pop_faction = {
						is_a_vanilla_faction = yes
						exists = leader
						leader = { not = { is_same_value = root.ruler } }
						faction_approval < 0.4
						support > 0.05
					}
				}
				random_pop_faction = {
					limit = {
						is_a_vanilla_faction = yes
						exists = leader
						leader = { not = { is_same_value = root.ruler } }
