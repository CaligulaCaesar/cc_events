#monthly
event = {
	id = ccrebel_perf.1
	is_triggered_only = yes
	hide_window = yes
	
	immediate = {
		every_country = {
			limit = {
				OR = {
					is_country_type = default
					is_country_type = awakened_fallen_empire
				}
			}
			if = {
				limit = {
					NAND = {
						OR = {
							has_country_resource = {
								type = minerals
								amount > 1000
							}
							has_monthly_income = {
								resource = minerals
								value > 50
							}
						}
						OR = {
							has_monthly_income = {
								resource = energy
								value > 50
							}
							resource_stockpile_compare = {
								resource = energy
								value > 1000
							}
						}
					}
				}
				set_timed_country_flag = {
					flag = ccrebel_poor
					days = 31
				}
			}
			else_if = {
				limit = {
					NAND = {
						OR = {
							has_country_resource = {
								type = minerals
								amount > 1800
							}
							has_monthly_income = {
								resource = minerals
								value > 90
							}
						}
						OR = {
							has_monthly_income = {
								resource = energy
								value > 90
							}
							resource_stockpile_compare = {
								resource = energy
								value > 1800
							}
						}
					}
				}
				set_timed_country_flag = {
					flag = ccrebel_medium_poor
					days = 31
				}
			}
			else_if = {
				limit = {
					NAND = {
						OR = {
							has_country_resource = {
								type = minerals
								amount > 2500
							}
							has_monthly_income = {
								resource = minerals
								value > 125
							}
						}
						OR = {
							has_monthly_income = {
								resource = energy
								value > 125
							}
							resource_stockpile_compare = {
								resource = energy
								value > 2500
							}
						}
					}
				}
				set_timed_country_flag = {
					flag = ccrebel_medium_income
					days = 31
				}
			}
			else_if = {
				limit = {
					owner = {
						OR = {
							has_country_resource = {
								type = minerals
								amount > 5000
							}
							has_monthly_income = {
								resource = minerals
								value > 250
							}
						}
						OR = {
							has_monthly_income = {
								resource = energy
								value > 250
							}
							resource_stockpile_compare = {
								resource = energy
								value > 5000
							}
						}
					}
				}
				set_timed_country_flag = {
					flag = ccrebel_medium_rich
					days = 31
				}
				if = {
					limit = {
						owner = {
							OR = {
								has_country_resource = {
									type = minerals
									amount > 10000
								}
								has_monthly_income = {
									resource = minerals
									value > 500
								}
							}
							OR = {
								has_monthly_income = {
									resource = energy
									value > 500
								}
								resource_stockpile_compare = {
									resource = energy
									value > 10000
								}
							}
						}
					}
					set_timed_country_flag = {
						flag = ccrebel_rich
						days = 31
					}
				}
			}
		}
	}
}
