			}
			add_modifier = {
				modifier = dpe_missionary_planet
				days = -1
			}
		}

			NOR = {
				allows_slavery = yes
				has_ethic = ethic_xenophobe
				has_ethic = ethic_fanatic_xenophobe
			}
		}
