
namespace = dpe_init


country_event = {
	id = dpe_init.1000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type_with_subjects = yes
	}
	
	immediate = {
		if = {
			limit = { is_democracy = yes }
		}
		else_if = {
			limit = { is_imperial = yes }
		}
		else_if = {
			limit = { is_dictatorial = yes }
		}
		else_if = {
			limit = { has_authority = auth_oligarchic }
		}
		else_if = {
			limit = { is_megacorp = yes }
		}
	}
}
